1.두 정수를 입력하고 switch구문을 활용하여 사칙연산의 값을 구하시오.


void main(){
int a,b;
char c;
scanf("%d %d %d",&a,,&b,&c);
switch(c) {
case'+':
printf("%d,a+b);
break;
case'-':
printf("%d",a-b);
break;
case'/':
printf("%d,a/b);
break;
case'*':}
}
